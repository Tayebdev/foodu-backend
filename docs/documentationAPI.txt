
Class Name: User
Description: Represents a registered user of the system. This is the base entity for all roles (Admin, Client, RestaurantOwner, Driver).

Attributes:
  - userId: String (PK) - Unique identifier for the user
  - name: String - Full name of the user
  - email: String (Unique) - Email address for login and communication
  - phone: String (Unique) - Phone number for verification and contact
  - passwordHash: String - Hashed password for security
  - role: Enum{Admin, Client, RestaurantOwner, Driver} - User's role in the system
  - picture: String - URL to profile picture
  - verified: Boolean - Account verification status
  - isActive: Boolean - Account activation status (can be deactivated by user)
  - isBanned: Boolean - Account ban status (set by admin)
  - verifyCode: String - Code sent for email/phone verification
  - createdAt: Date - Account creation timestamp
  - updatedAt: Date - Last update timestamp
  - lastLogin: Date - Timestamp of last successful login

Methods / Operations:
  - register(email, password, role)
    Route: POST /auth/register
    Authorization: Public
    Description: Registers a new user with the specified role.
    Parameters:
      - email: String - User email
      - password: String - User password
      - role: String - Desired role (Client, RestaurantOwner, Driver)
    Request JSON Example:
      {
        "email": "user@example.com",
        "password": "securePass123",
        "role": "Client",
        "name": "John Doe",
        "phone": "+1234567890"
      }
    Success Response Example:
      {
        "userId": "u123",
        "token": "jwt_token_..."
      }
    Possible Errors:
      - 409: Email or phone already exists
      - 422: Invalid input data

  - login(email, password)
    Route: POST /auth/login
    Authorization: Public
    Description: Authenticates a user and returns a session token.
    Parameters:
      - email: String - User email
      - password: String - User password
    Request JSON Example:
      {
        "email": "user@example.com",
        "password": "securePass123"
      }
    Success Response Example:
      {
        "token": "jwt_token_...",
        "user": { "id": "u123", "role": "Client" }
      }
    Possible Errors:
      - 401: Invalid credentials
      - 403: Account banned or not verified

  - logout()
    Route: POST /auth/logout
    Authorization: Authenticated
    Description: Logs out the current user (invalidates token).
    Parameters: None
    Request JSON Example: {}
    Success Response Example: { "message": "Logged out successfully" }
    Possible Errors:
      - 401: Unauthorized

  - updateProfile(data)
    Route: PUT /users/profile
    Authorization: Authenticated
    Description: Updates the user's profile information.
    Parameters:
      - data: Object - Fields to update (name, picture, phone)
    Request JSON Example:
      {
        "name": "John Updated",
        "picture": "https://..."
      }
    Success Response Example:
      { "userId": "u123", "name": "John Updated", ... }
    Possible Errors:
      - 422: Invalid data

  - changePassword(oldPassword, newPassword)
    Route: PATCH /users/password
    Authorization: Authenticated
    Description: Changes the user's password.
    Parameters:
      - oldPassword: String - Current password
      - newPassword: String - New password
    Request JSON Example:
      {
        "oldPassword": "oldPass",
        "newPassword": "newPass"
      }
    Success Response Example: { "success": true }
    Possible Errors:
      - 401: Incorrect old password

  - resetPassword(token, newPassword)
    Route: POST /auth/reset-password
    Authorization: Public
    Description: Resets password using a verification token/code.
    Parameters:
      - token: String - Reset token
      - newPassword: String - New password
    Request JSON Example:
      { "token": "abc-123", "newPassword": "newPass" }
    Success Response Example: { "success": true }
    Possible Errors:
      - 400: Invalid or expired token

  - verifyEmail(token)
    Route: POST /auth/verify-email
    Authorization: Public
    Description: Verifies user's email address.
    Parameters:
      - token: String - Verification token
    Request JSON Example: { "token": "xyz-789" }
    Success Response Example: { "verified": true }
    Possible Errors:
      - 400: Invalid token

  - deleteAccount()
    Route: DELETE /users/account
    Authorization: Authenticated
    Description: Permanently deletes the user's account.
    Parameters: None
    Request JSON Example: {}
    Success Response Example: { "message": "Account deleted" }
    Possible Errors:
      - 400: Cannot delete account with active orders

  - getById(userId)
    Route: GET /users/:id
    Authorization: Admin
    Description: Retrieves a user by their ID.
    Parameters:
      - userId: String - The ID of the user
    Request JSON Example: N/A
    Success Response Example: { "userId": "u123", ... }
    Possible Errors:
      - 404: User not found

  - ban(reason)
    Route: PATCH /admin/users/:id/ban
    Authorization: Admin
    Description: Bans a user from the system.
    Parameters:
      - reason: String - Reason for the ban
    Request JSON Example: { "reason": "Violated terms" }
    Success Response Example: { "success": true, "isBanned": true }
